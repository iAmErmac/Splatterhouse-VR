Class TerrorBlast : Weapon //31988
{
	Default
	{
	//$Category Weapons
	Weapon.SelectionOrder 200;
	Weapon.SlotNumber 6;
	Weapon.Ammotype1 "cell";
	Weapon.AmmoUse1 1;
	Weapon.AmmoGive1 50;
	Weapon.Ammotype2 "cell";
	Weapon.AmmoUse2 0;
	Weapon.AmmoGive2 0;
	Weapon.Kickback 20;
	Inventory.Icon "cella0";
	Inventory.PickupMessage "You unlock the mask's inner powers!";
	Obituary "%k damned %o to hell.";
	}

	States
	{
	Spawn:
		SPIR ABCD 6 Bright;
		loop;
	Ready:
		RHAN A 1 A_WeaponReady;
		loop;
	Deselect:
		RHAN B 1 A_Lower;
		loop;
	Select: 
		RHAN B 1 A_Raise;
		loop;
	Fire:
		RHAN B 2 A_firecustommissile("HellBlast",0,1);
		RHAN B 2 A_CheckReload;
		goto Ready;
	AltFire:
		RBUT ABC 2;
		RBUT D 10 A_CustomPunch (20, FALSE, 0, "KickPuff", 96);
		RBUT EF 6;
		goto Ready;
	}
}
