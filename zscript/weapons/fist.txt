Class RickFuryFist : Fist replaces Fist
{
	Default
	{
	weapon.selectionorder 570;
	+WEAPON.NOALERT;
	Tag "PUNCH";
	}
	
	states
	{
	Select:
		FIST A 1 A_Raise(160);
		loop;
	Deselect:
		FIST A 1 A_Lower(160);
		loop;
	Ready:
		FIST A 1 A_WeaponReady;
		loop;
	Fire:
		FIST BC 1;
		TNT1 A 0 A_playsound("weapons/fist/swing");
		FIST DE 1;
		FIST E 0 A_jumpifinventory("PowerStrength",1,"PunchStrong");
		FIST E 3 A_Custompunch(10,0,1,"FuryFistPuff",96);
		FIST FGHI 1;
		FIST A 5;
		FIST A 0 A_Refire;
		FIST A 3;
		goto ready;
	PunchStrong:
		FIST BC 1;
		TNT1 A 0 A_playsound("weapons/fist/swing");
		FIST DE 1;
		RFIS E 3 A_Custompunch(40,0,1,"FistPuffStrong");
		FIST FGHI 1;
		FIST A 5;
		FIST A 0 A_Refire;
		FIST A 3;
		goto ready;
	AltFire:
		RBUT ABC 2;
		RBUT D 10 A_CustomPunch (20, FALSE, 0, "KickPuff", 96);
		RBUT EF 6;
		goto Ready;
	}
}

Class FuryFistPuff : Actor
{
	Default
	{
	+NOBLOCKMAP
	+NOGRAVITY
	+PUFFONACTORS
	-ALLOWPARTICLES
	+NOEXTREMEDEATH
	+PUFFONACTORS
	}
	
	States
	{
	Spawn:
		PUFF A 0;
		TNT1 A 1;
	Melee:
		TNT1 A 0 A_playsound("weapons/fist/hitwall", CHAN_WEAPON);
		PUFF AAAAAA 0 A_SpawnProjectile ("MeleeSmoke", 0, 0, random (0, 360), 2, random (0, 360));
		TNT1 A 1;
		Stop;
	XDeath:
		TNT1 A 0 A_playsound("weapons/fist/punch", CHAN_WEAPON);
		TNT1 A 1;
		Stop;
	}
}
